** pjsua/pjproject

https://docs.pjsip.org/en/latest/specific-guides/other/cli_cmd.html

Building:

#+begin_src bash

  sudo apt install build-essential libasound2-dev libssl-dev libncurses5-dev \
       libreadline-dev libavcodec-dev libavformat-dev libswscale-dev libopus-dev

  git clone https://github.com/pjsip/pjproject.git
  ./configure --enable-shared
  make dep
  make -j$(nproc)

  # apps are here and not installed
  cd pjsip-apps/src/pjsua


#+end_src

#+begin_src bash
  pjsua --id sip:6002@10.60.0.127 \
        --registrar sip:10.60.0.127 \
        --realm '*' \
        --username 6002 \
        --password 6002 \
        --local-port=5063
        # --use-tls

    pjsua --id sip:6001@10.60.0.127 \
        --registrar sip:10.60.0.127 \
        --realm '*' \
        --username 6001 \
        --password 6001 \
        --local-port=5062

#+end_src

#+begin_src bash
  # make call
  sip:6001@10.60.0.127;transport=udp
  sip:6002@10.60.0.127;transport=udp

  # answer with a
  # 200 → answer
  # 486 → busy
  # 603 → decline
#+end_src
